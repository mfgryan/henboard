const model = "items";
const INIT_ITEMS = "INIT_ITEMS";
function initItems(items) {
    return {
        type: INIT_ITEMS,
        model: model,
        skipValidation: true,
        items: items
    };
}
const OPEN_INFO = "OPEN_INFO";
function openInfo(item) {

}

const ADD_ITEM = "ADD_ITEM";
function addItem(item) {
    return {
        type: ADD_ITEM,
        insert: true,
        model: model,
        project: item.project,
        week: item.week || "",
        column: item.column || "",
        name: item.name,
        addItem: false
    };
}

    };
}
const MOVE_ITEM = "MOVE_ITEM";
function moveItem(item) {
    // locates an item by name and project
    // all other vars are what the matched item will be updated with
    return {
        type: MOVE_ITEM,
        model: model,
        project: item.project,
        name: item.name,
        week: item.week || "",
        column: item.column || ""
    };
}
const CHANGE_ITEM_VALUE = "CHANGE_ITEM_VALUE";
function changeItemValue(item) {

        project: item.project,
        name: item.name,
        value: item.value
    };
}
